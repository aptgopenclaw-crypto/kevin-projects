## 二、建議的設計：新增「步驟審核人設定表」

### 2.1 資料表設計

```sql
-- 流程步驟審核人設定表
CREATE TABLE workflow_step_assignees (
    id                  BIGSERIAL PRIMARY KEY,
    tenant_id           VARCHAR(32) NOT NULL,
    workflow_code       VARCHAR(64) NOT NULL,     -- 關聯流程定義
    step_id             VARCHAR(32) NOT NULL,     -- 關聯步驟定義
    role_code           VARCHAR(64) NOT NULL,     -- 對應步驟的 role_code（冗餘欄位，方便查詢）
    
    -- 指派類型
    assignee_type       VARCHAR(32) NOT NULL,     -- DEPARTMENT / ROLE / SPECIFIC_USER / DYNAMIC
    
    -- 當 assignee_type = DEPARTMENT 時：
    -- 從業務表單的哪個欄位取得部門 ID
    department_source   VARCHAR(64),               -- context.departmentId / business.specificDept
    
    -- 當 assignee_type = ROLE 時：
    -- 指定哪個角色（可跨部門）
    target_role_code    VARCHAR(64),               -- 如 ROLE_PROPERTY_MANAGER
    
    -- 當 assignee_type = SPECIFIC_USER 時：
    specific_user_id    VARCHAR(64),               -- 固定審核人
    
    -- 當 assignee_type = DYNAMIC 時：
    -- 呼叫哪個 Spring Bean 來動態解析
    resolver_bean       VARCHAR(128),              -- 如 "propertyManagerResolver"
    
    -- 排序（同一步驟有多個審核人時）
    priority            INT DEFAULT 0,
    
    created_at          TIMESTAMP DEFAULT NOW(),
    updated_at          TIMESTAMP DEFAULT NOW(),
    
    UNIQUE(tenant_id, workflow_code, step_id, priority)
);

CREATE INDEX idx_step_assignees ON workflow_step_assignees(tenant_id, workflow_code, step_id);
```

## 三、指派類型說明

| assignee_type | 說明 | 使用場景 | 範例 |
|---------------|------|---------|------|
| `DEPARTMENT` | 根據業務表單的部門欄位，查詢該部門的特定角色 | 部門主管審核 | step_manager |
| `ROLE` | 查詢某個角色的所有人（可跨部門） | 財產管理組、法務組 | step_property |
| `SPECIFIC_USER` | 固定指派給特定用戶 | 機關首長、監造人員 | CEO、監工 |
| `DYNAMIC` | 呼叫 Spring Bean 動態解析 | 申請人、巡檢排班表 | step_applicant |

## 五、更新 IAssigneeResolver 實作

```java
@Component
public class ConfigurableAssigneeResolver implements IAssigneeResolver {
    
    @Autowired
    private WorkflowStepAssigneeRepository assigneeRepo;
    
    @Autowired
    private OrganizationRepository orgRepo;
    
    @Autowired
    private ApplicationContext applicationContext;
    
    @Override
    public String resolve(StepDefinition stepDef, WorkflowContext context) {
        String tenantId = context.getTenantId();
        String workflowCode = context.getWorkflowCode();
        String stepId = stepDef.getId();
        
        // 1. 查詢步驟的審核人設定
        List<WorkflowStepAssignee> assignees = assigneeRepo.findByTenantIdAndWorkflowCodeAndStepId(
            tenantId, workflowCode, stepId
        );
        
        if (assignees.isEmpty()) {
            throw new AssigneeNotFoundException(stepId, context);
        }
        
        // 2. 依優先級嘗試解析
        for (WorkflowStepAssignee config : assignees) {
            String assignee = resolveByType(config, context);
            if (assignee != null) {
                // 3. 查詢代理人
                return applyDelegate(assignee, context);
            }
        }
        
        throw new AssigneeNotFoundException(stepId, context);
    }
    
    private String resolveByType(WorkflowStepAssignee config, WorkflowContext context) {
        return switch (config.getAssigneeType()) {
            case "DEPARTMENT" -> {
                String departmentId = getDepartmentFromContext(config.getDepartmentSource(), context);
                yield orgRepo.findDepartmentAdmin(context.getTenantId(), departmentId);
            }
            case "ROLE" -> orgRepo.findUsersByRole(context.getTenantId(), config.getTargetRoleCode())
                .stream().findFirst().orElse(null);
            case "SPECIFIC_USER" -> config.getSpecificUserId();
            case "DYNAMIC" -> {
                DynamicAssigneeResolver resolver = applicationContext.getBean(
                    config.getResolverBean(), DynamicAssigneeResolver.class
                );
                yield resolver.resolve(null, context);
            }
            default -> null;
        };
    }
}
```
